#Q.3Write a program for Bitwise operators 
#ans:
# Bitwise Operators

d = 10  # Binary: 1010
e = 4   # Binary: 0100

# Bitwise AND
bitwise_and = d &  e
print("Bitwise AND:", bitwise_and)  
# Bitwise OR
bitwise_or = d | e
print("Bitwise OR:", bitwise_or)  

# Bitwise XOR
bitwise_xor = d ^ e
print("Bitwise XOR:", bitwise_xor)  

# Bitwise NOT
bitwise_not = ~d
print("Bitwise NOT:", bitwise_not)  

# Bitwise left shift
left_shift = d << 2
print("Bitwise Left Shift:", left_shift)  

# Bitwise right shift
right_shift = d >> 2
print("Bitwise Right Shift:", right_shift)

output:
Bitwise AND: 0
Bitwise OR: 14
Bitwise XOR: 14
Bitwise NOT: -11
Bitwise Left Shift: 40    
Bitwise Right Shift: 2 


#Q.4 Write a program to calculate greatest of three numbers
#ans:
def greatest(a, b, c):
    if a == b == c:
        print("All three numbers are equal.")
    elif a == b:
        if a > c:
            print(f"{a} and {b} are equal and greater than {c}.")
        else:
            print(f"The greatest number is {c}, but other numbers are equal.")
    elif a == c:
        if a > b:
            print(f"{a} and {c} are equal and greater than {b}.")
        else:
            print(f"The greatest number is {b}, but other numbers are equal.")
    elif b == c:
        if b > a:
            print(f" b and c  are equal and greater than {a}.")
        else:
            print(f"The greatest number is {a}, but other numbers are equal.")
    else:
        if a > b and a > c:
            print(f"The greatest number is {a}.")
        elif b > a and b > c:
            print(f"The greatest number is {b}.")
        else:
            print(f"The greatest number is {c}.")

a = int(input("Enter the first number: "))
b = int(input("Enter the second number: "))
c = int(input("Enter the third number: "))

greatest(a, b, c)

output:
Enter the first number: 56
Enter the second number: 56
Enter the third number: 78
The greatest number is 78, but other numbers are equal.
